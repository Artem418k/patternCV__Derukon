import numpy as np
import matplotlib.pyplot as plt
from sklearn.tree import DecisionTreeClassifier, plot_tree

#  
x_train_5 = np.array([[27, 46], [8, 22], [5, 30], [5, 19], [15, 23], [34, 42], [10, 23], [35, 32]])
y_train_5 = np.array([-1, -1, 1, -1, 1, 1, -1, -1])

#    
plt.figure(figsize=(8, 6))
plt.xlim(0, 50)
plt.ylim(0, 50)
plt.fill_between([0, 50], 0, 50, color='lightgray')
plt.scatter(x_train_5[:, 0], x_train_5[:, 1], c=y_train_5, cmap='viridis', s=100, edgecolors='black', linewidth=1.5)
plt.title('   ')
plt.xlabel(' 1')
plt.ylabel(' 2')
plt.show()

#       
clf_tree = DecisionTreeClassifier(criterion='entropy', max_depth=3, random_state=5)
clf_tree.fit(x_train_5, y_train_5)

#     
tree_structure = plot_tree(clf_tree, filled=True, feature_names=['Feature 1', 'Feature 2'], class_names=['Class -1', 'Class 1'], fontsize=8)
plt.show()

#      
text_representation = tree_structure[0]
print(text_representation)
