import numpy as np
import matplotlib.pyplot as plt

#   
def loss(w, x, y):
    M = np.dot(w, x) * y
    return 2 / (1 + np.exp(M))

#        w
def df(w, x, y):
    L1 = 1.0  #  L1-
    M = np.dot(w, x) * y
    return -2 * (1 + np.exp(M)) ** (-2) * np.exp(M) * x * y + L1 * np.sign(w)

# 
x_train_5 = np.array([[42, 6], [39, 18], [25, 16], [27, 13], [24, 33], [8, 44], [30, 17], [7, 48], [25, 17]])
x_train_5 = np.hstack((x_train_5, np.ones((len(x_train_5), 1))))  #   
y_train_5 = np.array([1, 1, -1, -1, -1, -1, 1, -1, -1])

fn = len(x_train_5[0])
n_train = len(x_train_5)  #   
w = np.zeros(fn)           #   
nt = 0.00001               #   SGD
lm = 0.01                  #  ""  Q

    w = w - nt * df(w, x_train_5[k], y_train_5[k])  #     SGD
    Q = lm * ek + (1 - lm) * Q                  #
